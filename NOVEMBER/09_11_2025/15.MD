class Solution:
    def threeSum(self, nums: List[int]) -> List[List[int]]:
        # for each number in the string - run a binary search to find two numbers such that the sum of those two equal
        res = []
        nums.sort()
        for start in range(len(nums)):
            if start > 0 and nums[start] == nums[start - 1]:
                continue
            target = -nums[start]
            l = start + 1
            r = len(nums) - 1
            while l < r:
                two_sum = nums[l] + nums[r]
                if two_sum > target:
                    r -= 1
                elif two_sum < target:
                    l += 1
                else:
                    res.append([nums[start], nums[l], nums[r]])
                    l += 1
                    while l<r and nums[l] == nums[l-1]:
                        l += 1
        return res
